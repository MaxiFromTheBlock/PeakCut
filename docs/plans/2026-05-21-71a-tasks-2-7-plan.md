```python
def get_speech_audio_segment(session, start_ms, end_ms):
    """Zentrale Audio-Quellen-Wahl für Sprach-Wiedergabe und -Export.

    Regel:
    - Mix in mic_tracks UND in mic_audios geladen → nur Mix-Segment.
    - Sonst Overlay der echten Mics, soweit geladen.
    - Keine Audioquelle geladen → None.
    """
    mic_tracks = session.project.mic_tracks
    mic_audios = session.mic_audios

    mix_path = get_mix_track(session.project)
    if mix_path:
        try:
            idx = mic_tracks.index(mix_path)
        except ValueError:
            idx = None
        if idx is not None and idx < len(mic_audios):
            return mic_audios[idx][start_ms:end_ms]

    real_mic_indices = [
        i for i, p in enumerate(mic_tracks)
        if not is_mix_track(p) and i < len(mic_audios)
    ]
    if not real_mic_indices:
        return None
    segment = mic_audios[real_mic_indices[0]][start_ms:end_ms]
    for i in real_mic_indices[1:]:
        segment = segment.overlay(mic_audios[i][start_ms:end_ms])
    return segment
```
