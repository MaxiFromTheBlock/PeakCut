```python
def get_speech_audio_segment(session, start_ms, end_ms):
    """Zentrale Audio-Quellen-Wahl für Sprach-Wiedergabe und -Export.

    Regel:
    - mic_tracks/mic_audios müssen gleiche Länge haben (Pre-Condition).
      Sonst → None (korrupter Zustand, kein Silent-Fallback).
    - Mix in mic_tracks → nur Mix-Segment.
    - Sonst Overlay aller echten Mics.
    - Keine Audioquelle → None.
    """
    mic_tracks = session.project.mic_tracks
    mic_audios = session.mic_audios

    # Carl P2-Fix: Length-Mismatch ist korrupter Zustand, KEINE
    # halbe Recovery (würde sonst Mix wegfallen lassen ohne Hinweis).
    if len(mic_tracks) != len(mic_audios):
        return None

    mix_path = get_mix_track(session.project)
    if mix_path:
        try:
            idx = mic_tracks.index(mix_path)
            return mic_audios[idx][start_ms:end_ms]
        except ValueError:
            pass  # Sollte nicht passieren — get_mix_track liest aus mic_tracks

    real_mic_indices = [
        i for i, p in enumerate(mic_tracks) if not is_mix_track(p)
    ]
    if not real_mic_indices:
        return None
    segment = mic_audios[real_mic_indices[0]][start_ms:end_ms]
    for i in real_mic_indices[1:]:
        segment = segment.overlay(mic_audios[i][start_ms:end_ms])
    return segment
```
